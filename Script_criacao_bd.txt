CREATE TABLE `clickvans`.`usuario` (
  `ID` INT NOT NULL,
  `NOME` VARCHAR(100) NOT NULL,
  `EMAIL` VARCHAR(100) NULL,
  `SENHA` VARCHAR(45) NULL,
  `DATA_CADASTRO` DATETIME NULL,
  PRIMARY KEY (`ID`),
  UNIQUE INDEX `ID_UNIQUE` (`ID` ASC) VISIBLE);


CREATE TABLE `clickvans`.`cidade` (
  `ID` INT NOT NULL,
  `NOME` VARCHAR(100) NOT NULL,
  PRIMARY KEY (`ID`),
  UNIQUE INDEX `ID_UNIQUE` (`ID` ASC) VISIBLE);


CREATE TABLE `clickvans`.`cliente` (
  `ID` INT NOT NULL,
  `NOME` VARCHAR(200) NOT NULL,
  `CPF` VARCHAR(45) NOT NULL,
  `DATA_NASCIMENTO` DATETIME NOT NULL,
  `EMAIL` VARCHAR(100) NOT NULL,
  `CONTATO` VARCHAR(45) NOT NULL,
  PRIMARY KEY (`ID`),
  UNIQUE INDEX `ID_UNIQUE` (`ID` ASC) VISIBLE);


CREATE TABLE `clickvans`.`rota` (
  `ID` INT NOT NULL,
  `LOCAL_PARTIDA` INT NOT NULL,
  `LOCAL_CHEGADA` INT NOT NULL,
  PRIMARY KEY (`ID`),
  UNIQUE INDEX `ID_UNIQUE` (`ID` ASC) VISIBLE,
  INDEX `ID_idx` (`LOCAL_PARTIDA` ASC) VISIBLE,
  INDEX `local_chegada_pk_idx` (`LOCAL_CHEGADA` ASC) VISIBLE,
  CONSTRAINT `local_partida_pk`
    FOREIGN KEY (`LOCAL_PARTIDA`)
    REFERENCES `clickvans`.`cidade` (`ID`)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION,
  CONSTRAINT `local_chegada_pk`
    FOREIGN KEY (`LOCAL_CHEGADA`)
    REFERENCES `clickvans`.`cidade` (`ID`)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION);


CREATE TABLE `clickvans`.`preco_passagem` (
  `id` INT NOT NULL,
  `LOCAL_PARTIDA` INT NOT NULL,
  `LOCAL_CHEGADA` INT NOT NULL,
  `PRECO` DECIMAL(10,2) NOT NULL,
  `DURACAO_VIAGEM` DECIMAL(10,2) NOT NULL,
  PRIMARY KEY (`id`),
  UNIQUE INDEX `id_UNIQUE` (`id` ASC) VISIBLE,
  INDEX `LOCAL_CHEGADA_PK_idx` (`LOCAL_CHEGADA` ASC) VISIBLE,
  INDEX `LOCAL_PARTIDA_idx` (`LOCAL_PARTIDA` ASC) VISIBLE,
  CONSTRAINT `LOCAL_CHEGADA_CIDADE_PK`
    FOREIGN KEY (`LOCAL_CHEGADA`)
    REFERENCES `clickvans`.`cidade` (`ID`)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION,
  CONSTRAINT `LOCAL_PARTIDA_CIDADE_PK`
    FOREIGN KEY (`LOCAL_PARTIDA`)
    REFERENCES `clickvans`.`cidade` (`ID`)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION);
